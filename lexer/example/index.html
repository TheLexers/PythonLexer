def greetSomeone(name):
    print("Hello " + name + "!")

def factorial(num):
    # Implementation using Tail Recursion btw ;)
    def doFactorial(num, a):
        if (num == 0):
            return a
        else:
            return doFactorial(num - 1, a * num)
    
    return doFactorial(num, 1)
